Enable Write-Back to Google Sheets

To allow the dashboard to add, edit, and delete tasks directly in your Google Sheet, set up a Google Apps Script Web App. Follow these steps:

  1. Open your Google Sheet
  2. Click Extensions → Apps Script
  3. Delete any existing code and paste the script below
  4. Click Deploy → New Deployment → Web App
  5. Set Execute as: Me and Who has access: Anyone
  6. Click Deploy and copy the Web App URL
  7. Paste the URL in the box below and click Save
const SHEET_NAME = 'Sheet1'; function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'sync') { const rows = data.tasks.map(t => [ t.id,t.name,t.dept,t.committee,t.nbto,t.status, t.br,t.brdate,t.memo,t.contact,t.impact, t.started,t.target,t.drive,t.notes,t.order ]); sheet.clearContents(); sheet.appendRow(['id','name','dept','committee','nbto','status','br','brdate','memo','contact','impact','started','target','drive','notes','order']); if (rows.length > 0) sheet.getRange(2,1,rows.length,16).setValues(rows); return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({status:'error',msg:'Unknown action'})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); }
